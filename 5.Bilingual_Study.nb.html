This chunk loads the original Study 4 dataset and a translation table for Indian regions. It then reshapes the data to align English and Hindi response columns, matches language tags, recodes variables, and merges geographic state-level information.

SES Analysis

This section compares English vs. Hindi performance differences by subjective SES. It creates a median split for SES, calculates groupwise means, and then uses bootstrapping to compute standard errors around those means.

df_ses <- df4_monetary %>%
  mutate(Median_Split_SES = ifelse(SubjectiveSES <= median(SubjectiveSES, na.rm = TRUE),
                                   "Below Median", "Above Median"))

# Function to calculate bootstrapped SEs
bootstrap_se <- function(data, indices) {
  d <- data[indices, ]
  mean(d$diff, na.rm = TRUE)
}

# Calculate true group means and differences
ses_diff <- df_ses %>%
  filter(!is.na(Median_Split_SES)) %>%
  group_by(Median_Split_SES, Language) %>%
  summarise(mean_ans = mean(num_total_answers, na.rm = TRUE), .groups = "drop") %>%
  pivot_wider(names_from = Language, values_from = mean_ans) %>%
  mutate(diff = English - Hindi)

# Bootstrap only SEs
ses_boot <- df_ses %>%
  filter(Language %in% c("English", "Hindi")) %>%
  group_split(Median_Split_SES) %>%
  map_dfr(function(group) {
    cat_name <- unique(group$Median_Split_SES)
    group_wide <- group %>%
      group_by(Language) %>%
      summarise(mean = mean(num_total_answers, na.rm = TRUE), .groups = "drop") %>%
      pivot_wider(names_from = Language, values_from = mean)
    true_diff <- group_wide$English - group_wide$Hindi
    b <- boot(data = group, statistic = function(d, i) {
      sample <- d[i, ]
      mean(sample$num_total_answers[sample$Language == "English"], na.rm = TRUE) -
        mean(sample$num_total_answers[sample$Language == "Hindi"], na.rm = TRUE)
    }, R = 1000)
    tibble(
      Median_Split_SES = cat_name,
      diff = true_diff,
      se_diff = sd(b$t)
    )
  })

Testing Statistical Difference for SES

This chunk tests whether the effect of language (English vs. Hindi) on total answers differs significantly between subjective SES groups. We implement a bootstrap procedure to estimate the standard error and 95% confidence interval of the difference-in-differences (DiD) — the gap in language effects between “Above Median” and “Below Median” SES groups. A statistically significant result would indicate that SES moderates the language-performance relationship.


# Filter and prepare data
df_did <- df_ses %>%
  filter(Language %in% c("English", "Hindi"), !is.na(Median_Split_SES))

# Bootstrap function for DiD
did_bootstrap <- function(data, indices) {
  d <- data[indices, ]
  d_above <- d %>% filter(Median_Split_SES == "Above Median")
  d_below <- d %>% filter(Median_Split_SES == "Below Median")
  
  diff_above <- mean(d_above$num_total_answers[d_above$Language == "English"], na.rm = TRUE) -
                mean(d_above$num_total_answers[d_above$Language == "Hindi"], na.rm = TRUE)
  
  diff_below <- mean(d_below$num_total_answers[d_below$Language == "English"], na.rm = TRUE) -
                mean(d_below$num_total_answers[d_below$Language == "Hindi"], na.rm = TRUE)
  
  return(diff_above - diff_below)
}

# Run bootstrap
set.seed(123)
library(boot)
did_result <- boot(data = df_did, statistic = did_bootstrap, R = 1000)

# Confidence interval
did_ci <- boot.ci(did_result, type = "perc")$percent[4:5]

# Output
list(
  diff_in_diff = mean(did_result$t),
  se = sd(did_result$t),
  ci_lower = did_ci[1],
  ci_upper = did_ci[2],
  significant = ifelse(did_ci[1] > 0 | did_ci[2] < 0, "Yes", "No")
)

Income Analysis

This section bootstraps the standard errors of English-Hindi differences in number of answers by household income category, excluding non-working, student, and retired participants.

# Exclude non-working/student/retired categories
excluded_levels <- c("Income: Not working", "Income: Student", "Income: Retired", "Income: ₹15 lakhs and above")

df_income <- df4_monetary %>%
  filter(!(HHIncome %in% excluded_levels),
         Language %in% c("English", "Hindi"),
         !is.na(num_total_answers)) %>%
  mutate(HHIncome = factor(HHIncome, ordered = TRUE))

# Define bootstrap_diff function
bootstrap_diff <- function(data, indices) {
  d <- data[indices, ]
  mean(d$num_total_answers[d$Language == "English"], na.rm = TRUE) -
    mean(d$num_total_answers[d$Language == "Hindi"], na.rm = TRUE)
}

# Bootstrap by each income level
income_levels <- levels(df_income$HHIncome)

income_boot <- lapply(income_levels, function(level) {
  group_data <- df_income %>% filter(HHIncome == level)
  if (nrow(group_data) >= 10 && length(unique(group_data$Language)) == 2) {
    # True mean difference
    true_diff <- mean(group_data$num_total_answers[group_data$Language == "English"], na.rm = TRUE) -
      mean(group_data$num_total_answers[group_data$Language == "Hindi"], na.rm = TRUE)
    
    # Bootstrap SE
    b <- boot(data = group_data, statistic = bootstrap_diff, R = 1000)
    
    tibble(
      HHIncome = level,
      diff = true_diff,
      se_diff = sd(b$t)
    )
  } else {
    tibble(HHIncome = level, diff = NA, se_diff = NA)
  }
})

Regional Analysis

This section bootstraps English-Hindi differences by Indian state, grouping small samples as “Other”, and ranks them by distance to the equator.

focus_states <- c("Delhi", "Haryana", "Rajasthan", "Uttar Pradesh", "Bihar")

df_region <- df4_monetary %>%
  mutate(State_Grouped = if_else(Living_State %in% focus_states, Living_State, "Other")) %>%
  filter(Language %in% c("English", "Hindi"), !is.na(num_total_answers))

states <- unique(df_region$State_Grouped)

region_boot <- lapply(states, function(state) {
  data_state <- df_region %>% filter(State_Grouped == state)
  if (length(unique(data_state$Language)) == 2 && nrow(data_state) >= 10) {
    b <- boot(data = data_state, statistic = bootstrap_diff, R = 1000)
    tibble(State = state, diff = mean(data_state$num_total_answers[data_state$Language == "English"], na.rm = TRUE) -
                         mean(data_state$num_total_answers[data_state$Language == "Hindi"], na.rm = TRUE),
           se_diff = sd(b$t))
  } else {
    tibble(State = state, diff = NA, se_diff = NA)
  }
})
